## 답: 
내 인생 첫 greedy 문제여서  
문제 자체는 완죠니 어렵지는 않았지만.. 그냥.. 새로워서.. 어려웠따.....  


```c++

#include <string>
#include <vector>
#include <iostream>
using namespace std;

int solution(int n, vector<int> lost, vector<int> reserve) {
    int answer = 0;
    vector<int>student(n,0);
    
    for(int i=0; i<lost.size(); i++){
        student[lost[i]-1]--;
    }
    
    for(int i=0; i<reserve.size(); i++){
        student[reserve[i]-1]++;
    }
    // student{1,-1,1,-1,1}
    for(int i=0; i<n; i++){
        cout<<student[i]<< " ";
    }
    for(int i=0; i<lost.size() ; i++){
        int cur = lost[i]-1;
        if(student[cur] != -1) continue;
        
        if(student[cur-1]==1){
            student[cur]=0;
            student[cur-1]=0;
            continue;
        }
        if(student[cur+1]==1){
            student[cur]=0;
            student[cur+1]=0;
        }
    }
    
    for(int i=0; i<n; i++){
        cout<<"\n" << student[i]<< " ";
        if(student[i]!= -1) answer++;
    }
    
    return answer;
}
```
